import javax.swing.JOptionPane; public class WordCounter { public static void main(String[] args) { // Prompt the user to enter a sentence String sentence = JOptionPane.showInputDialog("Please enter a sentence:"); // Split the sentence into words based on spaces String[] wordsArray = sentence.split(" "); // Get the number of words by checking the length of the array int numberOfWords = wordsArray.length; // Display the number of words JOptionPane.showMessageDialog(null, "The number of words in your sentence is: " + numberOfWords); } }
